def find_file(parent_keyword: str, filename: str) -> Path:
    matches = [
        p
        for p in ROOT.rglob(filename)
        if parent_keyword in str(p.parent) and "metastasis_ready_methylation_analysis" not in str(p)
    ]
    if not matches:
        raise FileNotFoundError(f"Could not find {filename} under {parent_keyword}")
    return matches[0]


def parse_gene_attrs(attr: str) -> tuple[str | None, str | None, str | None]:
    gene_id = gene_name = gene_type = None
    for key, value in re.findall(r'([A-Za-z0-9_]+) "([^"]+)"', attr):
        if key == "gene_id":
            gene_id = value.split(".")[0]
        elif key == "gene_name":
            gene_name = value
        elif key in {"gene_type", "gene_biotype"}:
            gene_type = value
    return gene_id, gene_name, gene_type


def load_tss() -> pd.DataFrame:
    rows = []
    with gzip.open(GTF, "rt", encoding="utf-8", errors="replace") as handle:
        for line in handle:
            if line.startswith("#"):
                continue
            chrom, source, feature, start, end, score, strand, frame, attr = line.rstrip("\n").split("\t")
            if feature != "gene":
                continue
            gene_id, gene_name, gene_type = parse_gene_attrs(attr)
            if gene_name is None:
                continue
            start_i = int(start)
            end_i = int(end)
            tss = start_i if strand == "+" else end_i
            rows.append((chrom, tss, gene_id, gene_name, gene_type, strand))
    return pd.DataFrame(rows, columns=["chr", "tss", "gene_id", "gene_symbol", "gene_type", "strand"])


def annotate_promoters(promoter_file: Path, tss: pd.DataFrame, label: str) -> pd.DataFrame:
    regions = pd.read_csv(promoter_file, sep="\t")
    if not {"chr", "start", "end"}.issubset(regions.columns):
        raise ValueError(f"Missing chr/start/end in {promoter_file}")
    annotated = []
    for chrom, sub in regions.groupby("chr", sort=False):
        tss_chr = tss[tss["chr"] == chrom]
        if tss_chr.empty:
            continue
        starts = sub["start"].to_numpy()
        ends = sub["end"].to_numpy()
        for idx, row in sub.iterrows():
            overlap = tss_chr[(tss_chr["tss"] >= row["start"] - 500) & (tss_chr["tss"] <= row["end"] + 500)]
            if overlap.empty:
                distances = (tss_chr["tss"] - ((row["start"] + row["end"]) / 2)).abs()
                nearest_idx = distances.idxmin()
                nearest = tss_chr.loc[[nearest_idx]].copy()
                nearest["tss_distance"] = distances.loc[nearest_idx]
                overlap = nearest[nearest["tss_distance"] <= 5000]
            else:
                overlap = overlap.copy()
                overlap["tss_distance"] = (overlap["tss"] - ((row["start"] + row["end"]) / 2)).abs()
            for _, gene in overlap.iterrows():
                rec = row.to_dict()
                rec.update(
                    {
                        "direction": label,
                        "gene_symbol": gene["gene_symbol"],
                        "gene_id": gene["gene_id"],
                        "gene_type": gene["gene_type"],
                        "tss": gene["tss"],
                        "tss_distance": gene["tss_distance"],
                    }
                )
                annotated.append(rec)
    return pd.DataFrame(annotated)


def read_gene_txt(keyword: str) -> set[str]:
    genes = set()
    for p in ROOT.rglob("*.txt"):
        if keyword in str(p) and "metastasis_ready_methylation_analysis" not in str(p):
            for line in p.read_text(encoding="utf-8", errors="replace").splitlines():
                line = line.strip()
                if line:
                    genes.add(line)
    return genes


def main() -> None:
    RESULTS.mkdir(parents=True, exist_ok=True)
    tss = load_tss()
    tss.to_csv(RESULTS / "gencode_v44_gene_tss.tsv", sep="\t", index=False)

    promoter_down = find_file("差异启动子", "down_wilcox.bed")
    promoter_up = find_file("差异启动子", "up_wilcox.bed")
    hypo_annot = annotate_promoters(promoter_down, tss, "hypomethylated_promoter")
    hyper_annot = annotate_promoters(promoter_up, tss, "hypermethylated_promoter")
    promoter_annot = pd.concat([hypo_annot, hyper_annot], ignore_index=True)
    promoter_annot.to_csv(RESULTS / "MP_differential_promoter_gene_annotation.tsv", sep="\t", index=False)

    hypo_promoter = set(hypo_annot["gene_symbol"].dropna().astype(str))
    hyper_promoter = set(hyper_annot["gene_symbol"].dropna().astype(str))
    enhancer_low = read_gene_txt("低_增强子_基因")
    enhancer_high = read_gene_txt("高_增强子_基因")

    gene_sets = {
        "MP_promoter_hypomethylated": sorted(hypo_promoter),
        "MP_promoter_hypermethylated": sorted(hyper_promoter),
        "MP_manuscript_hypomethylated_candidates": sorted(MANUSCRIPT_HYPO | enhancer_low),
        "MP_manuscript_hypermethylated_candidates": sorted(MANUSCRIPT_HYPER | enhancer_high),
        "MP_core_TF": sorted(CORE_TF),
        "MP_scRNA_positive_signature": sorted((MANUSCRIPT_HYPO | enhancer_low | hypo_promoter | {"AR", "SP1", "GATA3", "GREB1"}) - MANUSCRIPT_HYPER),
        "MP_scRNA_negative_signature": sorted(MANUSCRIPT_HYPER | enhancer_high | hyper_promoter),
    }
    rows = []
    for name, genes in gene_sets.items():
        for gene in genes:
            rows.append({"gene_set": name, "gene_symbol": gene})
    pd.DataFrame(rows).to_csv(RESULTS / "MP_gene_sets.tsv", sep="\t", index=False)

    n_hypo_regions = pd.read_csv(promoter_down, sep="\t").shape[0]
    n_hyper_regions = pd.read_csv(promoter_up, sep="\t").shape[0]
    summary = [
        "# MP gene sets",
        "",
        f"- Hypomethylated promoter regions: {n_hypo_regions}",
        f"- Hypermethylated promoter regions: {n_hyper_regions}",
        f"- Annotated hypomethylated promoter genes: {len(hypo_promoter)}",
        f"- Annotated hypermethylated promoter genes: {len(hyper_promoter)}",
        f"- Manuscript hypomethylated/candidate genes: {len(MANUSCRIPT_HYPO | enhancer_low)}",
        f"- Manuscript hypermethylated/candidate genes: {len(MANUSCRIPT_HYPER | enhancer_high)}",
        f"- Core TF genes: {len(CORE_TF)}",
        "",
        "Outputs:",
        "- `MP_differential_promoter_gene_annotation.tsv`",
        "- `MP_gene_sets.tsv`",
    ]
    (RESULTS / "MP_gene_sets_summary.md").write_text("\n".join(summary) + "\n", encoding="utf-8")
    print("\n".join(summary))


if __name__ == "__main__":
    main()

def parse_sample(path: Path) -> dict[str, str]:
    name = path.name
    gsm = name.split("_")[0]
    patient_match = re.search(r"(PR\d+)", name)
    patient = patient_match.group(1) if patient_match else "unknown"
    condition = "Normal" if re.search(r"_N\d?_", name) else "Tumor"
    replicate_match = re.search(r"_(N\d?|T\d?)_", name)
    replicate = replicate_match.group(1) if replicate_match else "NA"
    return {"gsm": gsm, "sample": name.replace("_dge.txt.gz", ""), "patient": patient, "condition": condition, "replicate": replicate}

def read_selected_counts(path: Path, keep_genes: set[str]) -> tuple[pd.DataFrame, pd.DataFrame]:
    meta = parse_sample(path)
    df = pd.read_csv(path, sep="\t", compression="gzip", index_col=0)
    totals = df.sum(axis=0)
    detected = (df > 0).sum(axis=0)
    present = [g for g in keep_genes if g in df.index]
    sub = df.loc[present].copy()
    sub.columns = [f"{meta['sample']}|{c}" for c in sub.columns]
    cell_meta = pd.DataFrame(
        {
            "cell_id": sub.columns,
            "barcode": [c.split("|", 1)[1] for c in sub.columns],
            "sample": meta["sample"],
            "gsm": meta["gsm"],
            "patient": meta["patient"],
            "condition": meta["condition"],
            "replicate": meta["replicate"],
            "total_counts": totals.to_numpy(),
            "detected_genes": detected.to_numpy(),
        }
    )
    return sub, cell_meta


def log_normalize(counts: pd.DataFrame, totals: pd.Series) -> pd.DataFrame:
    aligned_totals = totals.reindex(counts.columns).replace(0, np.nan)
    norm = counts.div(aligned_totals, axis=1) * 1e4
    return np.log1p(norm).fillna(0)


def mean_score(expr: pd.DataFrame, genes: list[str]) -> pd.Series:
    present = [g for g in genes if g in expr.index]
    if not present:
        return pd.Series(np.nan, index=expr.columns)
    return expr.loc[present].mean(axis=0)


def assign_cell_types(expr: pd.DataFrame, meta: pd.DataFrame) -> pd.DataFrame:
    scores = pd.DataFrame(index=expr.columns)
    for label, genes in CELL_MARKERS.items():
        scores[label] = mean_score(expr, genes)
    best = scores.idxmax(axis=1)
    best_score = scores.max(axis=1)
    meta = meta.set_index("cell_id").copy()
    for label in scores.columns:
        meta[f"{label}_score"] = scores[label]
    meta["cell_type_broad"] = np.where(best_score >= 0.15, best, "Other")
    return meta.reset_index()


def compute_scores(expr: pd.DataFrame, meta: pd.DataFrame, gene_sets: dict[str, list[str]]) -> pd.DataFrame:
    meta = meta.set_index("cell_id").copy()
    pos = gene_sets.get("MP_scRNA_positive_signature", [])
    neg = gene_sets.get("MP_scRNA_negative_signature", [])
    manuscript_pos = gene_sets.get("MP_manuscript_hypomethylated_candidates", [])
    manuscript_neg = gene_sets.get("MP_manuscript_hypermethylated_candidates", [])
    core_tf = gene_sets.get("MP_core_TF", [])

    meta["MP_positive_score"] = mean_score(expr, pos)
    meta["MP_negative_score"] = mean_score(expr, neg)
    meta["MP_signature_score"] = meta["MP_positive_score"] - meta["MP_negative_score"]
    meta["MP_manuscript_positive_score"] = mean_score(expr, manuscript_pos)
    meta["MP_manuscript_negative_score"] = mean_score(expr, manuscript_neg)
    meta["MP_manuscript_score"] = meta["MP_manuscript_positive_score"] - meta["MP_manuscript_negative_score"]
    meta["MP_core_TF_score"] = mean_score(expr, core_tf)
    return meta.reset_index()


def statistics_table(meta: pd.DataFrame) -> pd.DataFrame:
    rows = []
    comparisons = [
        ("all_cells", meta),
        ("epithelial_like", meta[meta["cell_type_broad"] == "Epithelial"]),
    ]
    scores = ["MP_signature_score", "MP_manuscript_score", "MP_core_TF_score"]
    for subset_name, sub in comparisons:
        for score in scores:
            normal = sub.loc[sub["condition"] == "Normal", score].dropna()
            tumor = sub.loc[sub["condition"] == "Tumor", score].dropna()
            if len(normal) and len(tumor):
                stat, pvalue = stats.mannwhitneyu(tumor, normal, alternative="two-sided")
                effect = tumor.median() - normal.median()
            else:
                pvalue = np.nan
                effect = np.nan
            rows.append(
                {
                    "subset": subset_name,
                    "score": score,
                    "n_normal": len(normal),
                    "n_tumor": len(tumor),
                    "median_normal": normal.median() if len(normal) else np.nan,
                    "median_tumor": tumor.median() if len(tumor) else np.nan,
                    "median_difference_tumor_minus_normal": effect,
                    "pvalue_mannwhitney": pvalue,
                }
            )
    stats_df = pd.DataFrame(rows)
    p = stats_df["pvalue_mannwhitney"].to_numpy(dtype=float)
    ok = np.isfinite(p)
    fdr = np.full(len(p), np.nan)
    if ok.sum():
        order = np.argsort(p[ok])
        ranked = p[ok][order] * ok.sum() / np.arange(1, ok.sum() + 1)
        ranked = np.minimum.accumulate(ranked[::-1])[::-1]
        idx = np.where(ok)[0][order]
        fdr[idx] = np.clip(ranked, 0, 1)
    stats_df["FDR"] = fdr
    return stats_df


def prepare_embedding(expr: pd.DataFrame, meta: pd.DataFrame) -> pd.DataFrame:
    genes = expr.var(axis=1).sort_values(ascending=False).head(min(250, expr.shape[0])).index
    x = expr.loc[genes].T
    x = StandardScaler(with_mean=True, with_std=True).fit_transform(x)
    pcs = PCA(n_components=min(20, x.shape[1]), random_state=7).fit_transform(x)
    emb = umap.UMAP(n_neighbors=30, min_dist=0.35, random_state=7).fit_transform(pcs)
    out = meta[["cell_id"]].copy()
    out["UMAP1"] = emb[:, 0]
    out["UMAP2"] = emb[:, 1]
    return out

def parse_sample(path: Path) -> str:
    return path.name.replace("_dge.txt.gz", "")


def bh_fdr(pvalues: np.ndarray) -> np.ndarray:
    pvalues = np.asarray(pvalues, dtype=float)
    fdr = np.full(len(pvalues), np.nan)
    ok = np.isfinite(pvalues)
    if ok.sum() == 0:
        return fdr
    order = np.argsort(pvalues[ok])
    ranked = pvalues[ok][order] * ok.sum() / np.arange(1, ok.sum() + 1)
    ranked = np.minimum.accumulate(ranked[::-1])[::-1]
    out = np.empty(ok.sum())
    out[order] = np.clip(ranked, 0, 1)
    fdr[ok] = out
    return fdr

def add_rank_and_detection_scores(meta: pd.DataFrame, mp_genes: list[str]) -> pd.DataFrame:
    rows = []
    for path in sorted(RAW176.glob("*dge.txt.gz")):
        sample = parse_sample(path)
        sample_meta = meta[(meta["sample"].eq(sample)) & (meta["cell_type_broad"].eq("Epithelial"))].copy()
        if sample_meta.empty:
            continue
        counts = pd.read_csv(path, sep="\t", compression="gzip", index_col=0)
        barcodes = sample_meta["barcode"].tolist()
        barcodes = [b for b in barcodes if b in counts.columns]
        if not barcodes:
            continue
        sub = counts.loc[:, barcodes]
        present_mp = [g for g in mp_genes if g in sub.index]
        ranks = sub.rank(axis=0, method="average", pct=True)
        sample_scores = pd.DataFrame(
            {
                "cell_id": [f"{sample}|{b}" for b in barcodes],
                "MP_rank_score": ranks.loc[present_mp].mean(axis=0).to_numpy(),
                "MP_detected_fraction": (sub.loc[present_mp] > 0).mean(axis=0).to_numpy(),
                "n_MP_genes_present": len(present_mp),
            }
        )
        rows.append(sample_scores)
    score = pd.concat(rows, ignore_index=True)
    meta = meta.merge(score, on="cell_id", how="left")
    return meta


def proportion_analysis(meta: pd.DataFrame) -> tuple[pd.DataFrame, pd.DataFrame]:
    epi = meta[meta["cell_type_broad"].eq("Epithelial")].copy()
    normal = epi.loc[epi["condition"].eq("Normal"), "MP_rank_score"].dropna()
    thresholds = {
        "normal_q75": normal.quantile(0.75),
        "normal_q90": normal.quantile(0.90),
    }
    sample_rows = []
    test_rows = []
    for label, threshold in thresholds.items():
        epi[f"MP_high_{label}"] = epi["MP_rank_score"] >= threshold
        for (patient, sample, condition), sub in epi.groupby(["patient", "sample", "condition"]):
            n_total = int(sub[f"MP_high_{label}"].notna().sum())
            n_high = int(sub[f"MP_high_{label}"].sum())
            sample_rows.append(
                {
                    "threshold": label,
                    "threshold_value": threshold,
                    "patient": patient,
                    "sample": sample,
                    "condition": condition,
                    "n_epithelial_cells": n_total,
                    "n_MP_high": n_high,
                    "fraction_MP_high": n_high / n_total if n_total else np.nan,
                }
            )
        normal_counts = epi[epi["condition"].eq("Normal")][f"MP_high_{label}"].value_counts()
        tumor_counts = epi[epi["condition"].eq("Tumor")][f"MP_high_{label}"].value_counts()
        table = np.array(
            [
                [int(tumor_counts.get(True, 0)), int(tumor_counts.get(False, 0))],
                [int(normal_counts.get(True, 0)), int(normal_counts.get(False, 0))],
            ]
        )
        odds, p = stats.fisher_exact(table)
        test_rows.append(
            {
                "threshold": label,
                "threshold_value": threshold,
                "tumor_high": table[0, 0],
                "tumor_low": table[0, 1],
                "normal_high": table[1, 0],
                "normal_low": table[1, 1],
                "odds_ratio_tumor_vs_normal": odds,
                "pvalue_fisher": p,
            }
        )
    tests = pd.DataFrame(test_rows)
    tests["FDR"] = bh_fdr(tests["pvalue_fisher"].to_numpy())
    return pd.DataFrame(sample_rows), tests


def contribution_genes(meta: pd.DataFrame, mp_genes: list[str]) -> pd.DataFrame:
    expr = pd.read_pickle(PROC176 / "GSE176031_selected_log_normalized_expr.pkl")
    epi = meta[meta["cell_type_broad"].eq("Epithelial")].copy()
    genes = [g for g in mp_genes if g in expr.index]
    normal_cells = epi.loc[epi["condition"].eq("Normal"), "cell_id"]
    tumor_cells = epi.loc[epi["condition"].eq("Tumor"), "cell_id"]
    rows = []
    for gene in genes:
        n = expr.loc[gene, normal_cells]
        t = expr.loc[gene, tumor_cells]
        rows.append(
            {
                "gene": gene,
                "mean_normal": n.mean(),
                "mean_tumor": t.mean(),
                "delta_tumor_minus_normal": t.mean() - n.mean(),
                "pct_normal": (n > 0).mean() * 100,
                "pct_tumor": (t > 0).mean() * 100,
                "pct_delta": ((t > 0).mean() - (n > 0).mean()) * 100,
            }
        )
    out = pd.DataFrame(rows)
    out["support_score"] = out["delta_tumor_minus_normal"] * np.log1p(out["pct_tumor"])
    return out.sort_values("support_score", ascending=False)


def tumor_mp_high_low_gene_ranking(meta: pd.DataFrame, mp_genes: list[str]) -> tuple[pd.DataFrame, pd.DataFrame]:
    epi_tumor = meta[(meta["condition"].eq("Tumor")) & (meta["cell_type_broad"].eq("Epithelial"))].copy()
    hi = epi_tumor["MP_rank_score"].quantile(0.75)
    lo = epi_tumor["MP_rank_score"].quantile(0.25)
    high_cells = set(epi_tumor.loc[epi_tumor["MP_rank_score"] >= hi, "cell_id"])
    low_cells = set(epi_tumor.loc[epi_tumor["MP_rank_score"] <= lo, "cell_id"])

    sum_high = None
    sum_low = None
    det_high = None
    det_low = None
    n_high = 0
    n_low = 0
    genes_index = None
    for path in sorted(RAW176.glob("*dge.txt.gz")):
        sample = parse_sample(path)
        high_barcodes = [cid.split("|", 1)[1] for cid in high_cells if cid.startswith(sample + "|")]
        low_barcodes = [cid.split("|", 1)[1] for cid in low_cells if cid.startswith(sample + "|")]
        if not high_barcodes and not low_barcodes:
            continue
        counts = pd.read_csv(path, sep="\t", compression="gzip", index_col=0)
        genes_index = counts.index if genes_index is None else genes_index
        high_barcodes = [b for b in high_barcodes if b in counts.columns]
        low_barcodes = [b for b in low_barcodes if b in counts.columns]
        if sum_high is None:
            sum_high = pd.Series(0.0, index=counts.index)
            sum_low = pd.Series(0.0, index=counts.index)
            det_high = pd.Series(0.0, index=counts.index)
            det_low = pd.Series(0.0, index=counts.index)
        if high_barcodes:
            high = counts.loc[:, high_barcodes]
            totals = high.sum(axis=0).replace(0, np.nan)
            norm = np.log1p(high.div(totals, axis=1) * 1e4).fillna(0)
            sum_high = sum_high.add(norm.sum(axis=1), fill_value=0)
            det_high = det_high.add((high > 0).sum(axis=1), fill_value=0)
            n_high += len(high_barcodes)
        if low_barcodes:
            low = counts.loc[:, low_barcodes]
            totals = low.sum(axis=0).replace(0, np.nan)
            norm = np.log1p(low.div(totals, axis=1) * 1e4).fillna(0)
            sum_low = sum_low.add(norm.sum(axis=1), fill_value=0)
            det_low = det_low.add((low > 0).sum(axis=1), fill_value=0)
            n_low += len(low_barcodes)

    mean_high = sum_high / max(n_high, 1)
    mean_low = sum_low / max(n_low, 1)
    ranking = pd.DataFrame(
        {
            "gene": mean_high.index,
            "mean_MP_high": mean_high.to_numpy(),
            "mean_MP_low": mean_low.reindex(mean_high.index).to_numpy(),
            "logFC_MP_high_vs_low": (mean_high - mean_low.reindex(mean_high.index)).to_numpy(),
            "pct_MP_high": (det_high / max(n_high, 1) * 100).reindex(mean_high.index).to_numpy(),
            "pct_MP_low": (det_low / max(n_low, 1) * 100).reindex(mean_high.index).to_numpy(),
        }
    ).sort_values("logFC_MP_high_vs_low", ascending=False)
    ranking = ranking[np.isfinite(ranking["logFC_MP_high_vs_low"])].copy()
    ranking = ranking[~ranking["gene"].str.startswith(("MT-", "RPL", "RPS"), na=False)].copy()

    enrich_rows = []
    all_values = ranking.set_index("gene")["logFC_MP_high_vs_low"].dropna()
    for name, genes in PATHWAY_SETS.items():
        present = [g for g in genes if g in all_values.index]
        background = all_values.drop(index=present, errors="ignore")
        if len(present) >= 2:
            stat, p = stats.mannwhitneyu(all_values.loc[present], background, alternative="greater")
            enrich_rows.append(
                {
                    "pathway": name,
                    "n_present": len(present),
                    "mean_logFC_geneset": all_values.loc[present].mean(),
                    "median_logFC_geneset": all_values.loc[present].median(),
                    "pvalue_rank_enrichment": p,
                    "present_genes": ",".join(present),
                }
            )
    enrich = pd.DataFrame(enrich_rows)
    enrich["FDR"] = bh_fdr(enrich["pvalue_rank_enrichment"].to_numpy())
    return ranking, enrich.sort_values("mean_logFC_geneset", ascending=False)


def spatial_mp_high_low() -> tuple[pd.DataFrame, pd.DataFrame]:
    spots = pd.read_csv(RES230 / "GSE230282_spatial_spot_MP_scores.tsv", sep="\t")
    hi = spots["MP_positive_promoter_score"].quantile(0.75)
    lo = spots["MP_positive_promoter_score"].quantile(0.25)
    high = spots[spots["MP_positive_promoter_score"] >= hi].copy()
    low = spots[spots["MP_positive_promoter_score"] <= lo].copy()
    sigs = ["Epithelial", "AR_luminal", "EMT_plasticity", "Proliferation", "Neuroendocrine", "Stromal_ECM"]
    rows = []
    for sig in sigs:
        col = f"{sig}_score"
        stat, p = stats.mannwhitneyu(high[col].dropna(), low[col].dropna(), alternative="two-sided")
        rows.append(
            {
                "signature": sig,
                "mean_MP_high_spots": high[col].mean(),
                "mean_MP_low_spots": low[col].mean(),
                "median_MP_high_spots": high[col].median(),
                "median_MP_low_spots": low[col].median(),
                "median_difference_high_minus_low": high[col].median() - low[col].median(),
                "pvalue_mannwhitney": p,
                "n_high_spots": len(high),
                "n_low_spots": len(low),
            }
        )
    tests = pd.DataFrame(rows)
    tests["FDR"] = bh_fdr(tests["pvalue_mannwhitney"].to_numpy())
    plot = pd.concat(
        [
            high.assign(MP_group="MP-high spots"),
            low.assign(MP_group="MP-low spots"),
        ],
        ignore_index=True,
    )
    return plot, tests

def bh_fdr(pvalues: np.ndarray) -> np.ndarray:
    pvalues = np.asarray(pvalues, dtype=float)
    fdr = np.full(len(pvalues), np.nan)
    ok = np.isfinite(pvalues)
    if ok.sum() == 0:
        return fdr
    order = np.argsort(pvalues[ok])
    ranked = pvalues[ok][order] * ok.sum() / np.arange(1, ok.sum() + 1)
    ranked = np.minimum.accumulate(ranked[::-1])[::-1]
    out = np.empty(ok.sum())
    out[order] = np.clip(ranked, 0, 1)
    fdr[ok] = out
    return fdr

def matched_random_spatial_controls(gene_sets: dict[str, list[str]], n_iter: int = 500) -> tuple[pd.DataFrame, pd.DataFrame]:
    rng = np.random.default_rng(20260516)
    features, expr, _spots = load_spatial_expr()
    existing = pd.read_csv(RES230 / "GSE230282_spatial_spot_MP_scores.tsv", sep="\t")
    hypo = sorted(set(gene_sets["MP_promoter_hypomethylated"]) & set(features["gene_symbol"]))
    exclude = set(gene_sets["MP_promoter_hypomethylated"]) | set(gene_sets["MP_promoter_hypermethylated"])

    gene_means = np.asarray(expr.mean(axis=1)).ravel()
    gene_table = pd.DataFrame({"gene": features["gene_symbol"].astype(str), "mean_expr": gene_means})
    gene_table = gene_table.groupby("gene", as_index=False)["mean_expr"].mean()
    gene_table = gene_table[gene_table["mean_expr"] > 0].copy()
    gene_table["bin"] = pd.qcut(gene_table["mean_expr"].rank(method="first"), q=20, labels=False)
    gene_to_bin = dict(zip(gene_table["gene"], gene_table["bin"]))
    bin_to_genes = {
        b: sorted(set(sub["gene"]) - exclude)
        for b, sub in gene_table.groupby("bin")
    }

    score_frame = _spots[["barcode"]].copy()
    score_frame["observed_score"] = score_signature(expr, features, hypo)
    aligned = existing.merge(score_frame, on="barcode", how="left")
    observed_score = aligned["observed_score"].to_numpy()
    observed = {}
    for program in ["AR_luminal", "Proliferation", "Neuroendocrine"]:
        r, _p = stats.spearmanr(observed_score, aligned[f"{program}_score"])
        observed[program] = r

    random_rows = []
    for i in range(n_iter):
        sampled = []
        for gene in hypo:
            b = gene_to_bin.get(gene)
            pool = bin_to_genes.get(b, [])
            if not pool:
                continue
            sampled.append(rng.choice(pool))
        sampled = sorted(set(sampled))
        if len(sampled) < max(10, len(hypo) // 3):
            continue
        score_frame = _spots[["barcode"]].copy()
        score_frame["random_score"] = score_signature(expr, features, sampled)
        aligned_random = existing[["barcode", "AR_luminal_score", "Proliferation_score", "Neuroendocrine_score"]].merge(score_frame, on="barcode", how="left")
        random_score = aligned_random["random_score"].to_numpy()
        row = {"iteration": i, "n_genes": len(sampled)}
        for program in ["AR_luminal", "Proliferation", "Neuroendocrine"]:
            r, _p = stats.spearmanr(random_score, aligned_random[f"{program}_score"])
            row[f"{program}_spearman_r"] = r
        random_rows.append(row)
    random_df = pd.DataFrame(random_rows)
    random_df.to_csv(OUT / "GSE230282_matched_random_gene_set_correlations.tsv", sep="\t", index=False)

    summary_rows = []
    for program, obs in observed.items():
        rand = random_df[f"{program}_spearman_r"].dropna()
        if program == "AR_luminal":
            empirical = ((rand <= obs).sum() + 1) / (len(rand) + 1)
            direction = "lower_than_random"
        else:
            empirical = ((rand >= obs).sum() + 1) / (len(rand) + 1)
            direction = "higher_than_random"
        summary_rows.append(
            {
                "program": program,
                "observed_spearman_r": obs,
                "random_mean_r": rand.mean(),
                "random_sd_r": rand.std(),
                "random_q025": rand.quantile(0.025),
                "random_q975": rand.quantile(0.975),
                "empirical_pvalue": empirical,
                "test_direction": direction,
                "n_random_sets": len(rand),
            }
        )
    summary = pd.DataFrame(summary_rows)
    summary.to_csv(OUT / "GSE230282_matched_random_gene_set_summary.tsv", sep="\t", index=False)
    return random_df, summary


def scrna_hypo_hyper_control(gene_sets: dict[str, list[str]]) -> pd.DataFrame:
    expr = pd.read_pickle(PROC176 / "GSE176031_selected_log_normalized_expr.pkl")
    meta = pd.read_csv(RES176 / "GSE176031_cell_metadata_with_MP_scores.tsv", sep="\t")
    epi = meta[meta["cell_type_broad"].eq("Epithelial")].copy()
    scores = {}
    for label, genes in {
        "hypomethylated": gene_sets["MP_promoter_hypomethylated"],
        "hypermethylated": gene_sets["MP_promoter_hypermethylated"],
    }.items():
        present = [g for g in genes if g in expr.index]
        epi[f"{label}_score"] = expr.loc[present, epi["cell_id"]].mean(axis=0).to_numpy()
        scores[label] = present
    epi["hypo_minus_hyper_score"] = epi["hypomethylated_score"] - epi["hypermethylated_score"]
    rows = []
    for score in ["hypomethylated_score", "hypermethylated_score", "hypo_minus_hyper_score"]:
        normal = epi.loc[epi["condition"].eq("Normal"), score].dropna()
        tumor = epi.loc[epi["condition"].eq("Tumor"), score].dropna()
        stat, p = stats.mannwhitneyu(tumor, normal, alternative="two-sided")
        rows.append(
            {
                "score": score,
                "n_present_genes_hypo": len(scores["hypomethylated"]),
                "n_present_genes_hyper": len(scores["hypermethylated"]),
                "median_normal": normal.median(),
                "median_tumor": tumor.median(),
                "median_difference_tumor_minus_normal": tumor.median() - normal.median(),
                "pvalue_mannwhitney": p,
            }
        )
    out = pd.DataFrame(rows)
    out["FDR"] = bh_fdr(out["pvalue_mannwhitney"].to_numpy())
    out.to_csv(OUT / "GSE176031_epithelial_hypo_hyper_promoter_control.tsv", sep="\t", index=False)
    return out

def bh_fdr(pvalues: np.ndarray) -> np.ndarray:
    pvalues = np.asarray(pvalues, dtype=float)
    fdr = np.full(len(pvalues), np.nan)
    ok = np.isfinite(pvalues)
    if ok.sum() == 0:
        return fdr
    order = np.argsort(pvalues[ok])
    ranked = pvalues[ok][order] * ok.sum() / np.arange(1, ok.sum() + 1)
    ranked = np.minimum.accumulate(ranked[::-1])[::-1]
    out = np.empty(ok.sum())
    out[order] = np.clip(ranked, 0, 1)
    fdr[ok] = out
    return fdr

    return path.name.replace("_dge.txt.gz", "")

def scRNA_detectability_and_shortlist(hypo: list[str]) -> tuple[pd.DataFrame, pd.DataFrame]:
    meta = pd.read_csv(RES176 / "GSE176031_cell_metadata_with_MP_scores.tsv", sep="\t")
    epi = meta[meta["cell_type_broad"].eq("Epithelial")].copy()
    rows = []
    for path in sorted(RAW176.glob("*dge.txt.gz")):
        sample = parse_sample(path)
        sample_epi = epi[epi["sample"].eq(sample)].copy()
        if sample_epi.empty:
            continue
        counts = pd.read_csv(path, sep="\t", compression="gzip", index_col=0)
        genes = [g for g in hypo if g in counts.index]
        barcodes = [b for b in sample_epi["barcode"] if b in counts.columns]
        if not genes or not barcodes:
            continue
        sub = counts.loc[genes, barcodes]
        totals = counts.loc[:, barcodes].sum(axis=0).replace(0, np.nan)
        norm = np.log1p(sub.div(totals, axis=1) * 1e4).fillna(0)
        cell_condition = sample_epi.set_index("barcode").loc[barcodes, "condition"]
        for gene in genes:
            for condition in ["Normal", "Tumor"]:
                cells = cell_condition.index[cell_condition.eq(condition)]
                if len(cells) == 0:
                    continue
                rows.append(
                    {
                        "gene": gene,
                        "sample": sample,
                        "condition": condition,
                        "n_cells": len(cells),
                        "mean_expression": norm.loc[gene, cells].mean(),
                        "pct_expressing": (sub.loc[gene, cells] > 0).mean() * 100,
                    }
                )
    per_sample = pd.DataFrame(rows)
    summary = (
        per_sample.groupby(["gene", "condition"], as_index=False)
        .agg(mean_expression=("mean_expression", "mean"), pct_expressing=("pct_expressing", "mean"), n_cells=("n_cells", "sum"))
        .pivot(index="gene", columns="condition", values=["mean_expression", "pct_expressing"])
    )
    summary.columns = [f"{a}_{b}" for a, b in summary.columns]
    summary = summary.reset_index().fillna(0)
    summary["delta_mean_tumor_minus_normal"] = summary["mean_expression_Tumor"] - summary["mean_expression_Normal"]
    summary["delta_pct_tumor_minus_normal"] = summary["pct_expressing_Tumor"] - summary["pct_expressing_Normal"]
    summary["is_housekeeping_like"] = summary["gene"].str.startswith(HOUSEKEEPING_PREFIX) | summary["gene"].isin(HOUSEKEEPING_GENES)

    spatial = pd.read_csv(CTRL / "GSE230282_spatial_hypo_hyper_scores.tsv", sep="\t")
    # Contribution per gene in spatial state: correlation with hypo-minus-hyper-like axes
    # The summary below relies on scRNA detectability plus spatial-compatible filters.
    summary["shortlist_score"] = (
        summary["delta_mean_tumor_minus_normal"] * np.log1p(summary["pct_expressing_Tumor"])
        + 0.02 * summary["delta_pct_tumor_minus_normal"]
    )
    shortlist = summary[
        (~summary["is_housekeeping_like"])
        & (summary["pct_expressing_Tumor"] >= 1.0)
        & (summary["delta_mean_tumor_minus_normal"] > 0)
    ].sort_values("shortlist_score", ascending=False)
    return summary.sort_values("shortlist_score", ascending=False), shortlist.head(15)


def spatial_state_axis_and_epithelial_corr() -> tuple[pd.DataFrame, pd.DataFrame]:
    spots = pd.read_csv(RES230 / "GSE230282_spatial_spot_MP_scores.tsv", sep="\t")
    score = pd.read_csv(CTRL / "GSE230282_spatial_hypo_hyper_scores.tsv", sep="\t")
    # Use the hypo-minus-hyper contrast for the final state plot because it
    # controls for general promoter-gene expression.
    cols = ["barcode", "MP_hypomethylated_promoter_score", "MP_hypermethylated_promoter_score", "MP_hypo_minus_hyper_score"]
    spots = spots.merge(score[cols], on="barcode", how="left")
    spots["NE_minus_AR_state"] = spots["Neuroendocrine_score"] - spots["AR_luminal_score"]
    spots["MP_state_group"] = np.where(
        spots["MP_hypo_minus_hyper_score"] >= spots["MP_hypo_minus_hyper_score"].quantile(0.75),
        "MP-high",
        np.where(spots["MP_hypo_minus_hyper_score"] <= spots["MP_hypo_minus_hyper_score"].quantile(0.25), "MP-low", "Mid"),
    )

    epi_cut = spots["Epithelial_score"].median()
    epi = spots[spots["Epithelial_score"] >= epi_cut].copy()
    rows = []
    for score_col in ["MP_hypomethylated_promoter_score", "MP_hypo_minus_hyper_score"]:
        for program in ["AR_luminal", "Proliferation", "Neuroendocrine", "Stromal_ECM"]:
            ok = epi[[score_col, f"{program}_score"]].dropna()
            r, p = stats.spearmanr(ok[score_col], ok[f"{program}_score"])
            rows.append({"score": score_col, "program": program, "spearman_r": r, "pvalue": p, "n_spots": len(ok)})
    corr = pd.DataFrame(rows)
    corr["FDR"] = bh_fdr(corr["pvalue"].to_numpy())
    return spots, corr

def zscore_rows(df: pd.DataFrame) -> pd.DataFrame:
    return df.sub(df.mean(axis=1), axis=0).div(df.std(axis=1).replace(0, np.nan), axis=0)


def mean_signature(expr: pd.DataFrame, genes: list[str], samples: list[str]) -> pd.Series:
    present = [g for g in genes if g in expr.index]
    if not present:
        return pd.Series(np.nan, index=samples)
    return zscore_rows(expr.loc[present, samples]).mean(axis=0)

def bh_fdr(pvalues: np.ndarray) -> np.ndarray:
    pvalues = np.asarray(pvalues, dtype=float)
    fdr = np.full(len(pvalues), np.nan)
    ok = np.isfinite(pvalues)
    if ok.sum() == 0:
        return fdr
    order = np.argsort(pvalues[ok])
    ranked = pvalues[ok][order] * ok.sum() / np.arange(1, ok.sum() + 1)
    ranked = np.minimum.accumulate(ranked[::-1])[::-1]
    out = np.empty(ok.sum())
    out[order] = np.clip(ranked, 0, 1)
    fdr[ok] = out
    return fdr


def zscore_rows(df: pd.DataFrame) -> pd.DataFrame:
    return df.sub(df.mean(axis=1), axis=0).div(df.std(axis=1).replace(0, np.nan), axis=0)


def mean_signature(expr: pd.DataFrame, genes: list[str], samples: list[str]) -> pd.Series:
    present = [g for g in genes if g in expr.index]
    if not present:
        return pd.Series(np.nan, index=samples)
    return zscore_rows(expr.loc[present, samples]).mean(axis=0)


def load_expression(target_genes: set[str]) -> pd.DataFrame:
    expr = pd.read_csv(RAW / "TCGA_PRAD_HiSeqV2.gz", sep="\t", compression="gzip")
    expr = expr.rename(columns={expr.columns[0]: "gene"}).set_index("gene")
    expr = expr.loc[expr.index.intersection(target_genes)]
    expr = expr.apply(pd.to_numeric, errors="coerce")
    return expr


def parse_t_stage(value) -> str:
    if pd.isna(value):
        return np.nan
    value = str(value).upper()
    if value.startswith("T2"):
        return "T2"
    if value.startswith("T3") or value.startswith("T4"):
        return "T3/T4"
    return np.nan


def parse_n_stage(value) -> str:
    if pd.isna(value):
        return np.nan
    value = str(value).upper()
    if value.startswith("N0"):
        return "N0"
    if value.startswith("N1"):
        return "N1"
    return np.nan


def gleason_group(value) -> str:
    if pd.isna(value):
        return np.nan
    try:
        v = int(float(value))
    except ValueError:
        return np.nan
    if v <= 6:
        return "<=6"
    if v == 7:
        return "7"
    return ">=8"


def prep_clinical_scores() -> pd.DataFrame:
    score = pd.read_csv(PREV / "TCGA_PRAD_MP_methylation_expression_scores.tsv", sep="\t", index_col=0)
    clin = pd.read_csv(RAW / "TCGA_PRAD_clinicalMatrix.tsv", sep="\t")
    merged = score.merge(clin, left_index=True, right_on="sampleID", how="inner")
    merged = merged[merged["sample_type"].eq("Primary Tumor")].copy()
    merged["Gleason_group"] = merged["gleason_score"].map(gleason_group)
    merged["T_group"] = merged["pathologic_T"].map(parse_t_stage)
    merged["N_group"] = merged["pathologic_N"].map(parse_n_stage)
    merged["BCR_status"] = merged["biochemical_recurrence"].map({"YES": "BCR", "NO": "No BCR"})
    merged["BCR_event"] = merged["biochemical_recurrence"].map({"YES": 1, "NO": 0})
    merged["BCR_time_days"] = np.where(
        merged["BCR_event"].eq(1),
        pd.to_numeric(merged["days_to_first_biochemical_recurrence"], errors="coerce"),
        pd.to_numeric(merged["days_to_last_followup"], errors="coerce"),
    )
    merged["age"] = pd.to_numeric(merged["age_at_initial_pathologic_diagnosis"], errors="coerce")
    merged["gleason_numeric"] = pd.to_numeric(merged["gleason_score"], errors="coerce")
    merged["advanced_T"] = merged["T_group"].map({"T2": 0, "T3/T4": 1})
    merged["node_positive"] = merged["N_group"].map({"N0": 0, "N1": 1})
    return merged


def clinical_group_tests(df: pd.DataFrame) -> pd.DataFrame:
    tests = []
    group_specs = [
        ("Gleason_group", ["<=6", "7", ">=8"]),
        ("T_group", ["T2", "T3/T4"]),
        ("N_group", ["N0", "N1"]),
        ("BCR_status", ["No BCR", "BCR"]),
    ]
    for score_col in MP_SCORE_COLS:
        for group_col, order in group_specs:
            sub = df[[score_col, group_col]].dropna()
            sub = sub[sub[group_col].isin(order)]
            if sub[group_col].nunique() < 2:
                continue
            values = [sub.loc[sub[group_col].eq(g), score_col].dropna() for g in order if g in set(sub[group_col])]
            if len(values) == 2:
                stat, p = stats.mannwhitneyu(values[0], values[1], alternative="two-sided")
                test = "Mann-Whitney"
            else:
                stat, p = stats.kruskal(*values)
                test = "Kruskal-Wallis"
            tests.append(
                {
                    "score": score_col,
                    "clinical_group": group_col,
                    "test": test,
                    "statistic": stat,
                    "pvalue": p,
                    "n": len(sub),
                    "group_medians": "; ".join(f"{g}={sub.loc[sub[group_col].eq(g), score_col].median():.3g}" for g in order if g in set(sub[group_col])),
                    "group_counts": "; ".join(f"{g}={sub[group_col].eq(g).sum()}" for g in order if g in set(sub[group_col])),
                }
            )
    out = pd.DataFrame(tests)
    out["FDR"] = bh_fdr(out["pvalue"].to_numpy())
    return out.sort_values(["score", "FDR"])


def survival_tests(df: pd.DataFrame) -> tuple[pd.DataFrame, pd.DataFrame]:
    surv = df.dropna(subset=["BCR_time_days", "BCR_event"]).copy()
    surv = surv[surv["BCR_time_days"] > 0].copy()
    rows = []
    cox_rows = []
    for score_col in MP_SCORE_COLS:
        x = surv[score_col]
        cut = x.median()
        high = surv[score_col] >= cut
        lr = logrank_test(
            surv.loc[high, "BCR_time_days"],
            surv.loc[~high, "BCR_time_days"],
            event_observed_A=surv.loc[high, "BCR_event"],
            event_observed_B=surv.loc[~high, "BCR_event"],
        )
        rows.append(
            {
                "score": score_col,
                "cutpoint": "median",
                "cut_value": cut,
                "n_high": int(high.sum()),
                "n_low": int((~high).sum()),
                "events_high": int(surv.loc[high, "BCR_event"].sum()),
                "events_low": int(surv.loc[~high, "BCR_event"].sum()),
                "logrank_pvalue": lr.p_value,
            }
        )
        cdat = surv[["BCR_time_days", "BCR_event", score_col, "age", "gleason_numeric", "advanced_T"]].dropna().copy()
        if cdat["BCR_event"].sum() >= 5:
            cdat[score_col + "_z"] = (cdat[score_col] - cdat[score_col].mean()) / cdat[score_col].std()
            cph = CoxPHFitter()
            cph.fit(cdat[["BCR_time_days", "BCR_event", score_col + "_z"]], duration_col="BCR_time_days", event_col="BCR_event")
            s = cph.summary.loc[score_col + "_z"]
            cox_rows.append({"model": "univariate", "score": score_col, "HR_per_SD": s["exp(coef)"], "pvalue": s["p"], "n": len(cdat), "events": int(cdat["BCR_event"].sum())})
            cph2 = CoxPHFitter()
            covars = [score_col + "_z", "age", "gleason_numeric", "advanced_T"]
            cph2.fit(cdat[["BCR_time_days", "BCR_event"] + covars], duration_col="BCR_time_days", event_col="BCR_event")
            s2 = cph2.summary.loc[score_col + "_z"]
            cox_rows.append({"model": "adjusted_age_gleason_T", "score": score_col, "HR_per_SD": s2["exp(coef)"], "pvalue": s2["p"], "n": len(cdat), "events": int(cdat["BCR_event"].sum())})
    km = pd.DataFrame(rows)
    km["FDR"] = bh_fdr(km["logrank_pvalue"].to_numpy())
    cox = pd.DataFrame(cox_rows)
    cox["FDR"] = bh_fdr(cox["pvalue"].to_numpy())
    return km, cox


def signature_scores(samples: list[str]) -> tuple[pd.DataFrame, pd.DataFrame]:
    genes = {g for genes in SIGNATURES.values() for g in genes}
    expr = load_expression(genes)
    available_samples = [s for s in samples if s in expr.columns]
    rows = {}
    overlap = []
    for name, geneset in SIGNATURES.items():
        present = [g for g in geneset if g in expr.index]
        overlap.append({"signature": name, "n_genes_defined": len(geneset), "n_genes_present": len(present), "present_genes": ",".join(present)})
        rows[name] = mean_signature(expr, geneset, available_samples)
    sig = pd.DataFrame(rows)
    return sig, pd.DataFrame(overlap)


def signature_correlation(df: pd.DataFrame, sig: pd.DataFrame) -> pd.DataFrame:
    merged = df.set_index("sampleID")[MP_SCORE_COLS].join(sig, how="inner")
    rows = []
    for mp in MP_SCORE_COLS:
        for signature in SIGNATURES:
            ok = merged[[mp, signature]].dropna()
            r, p = stats.spearmanr(ok[mp], ok[signature])
            rows.append({"mp_score": mp, "signature": signature, "spearman_r": r, "pvalue": p, "n_samples": len(ok)})
    out = pd.DataFrame(rows)
    out["FDR"] = bh_fdr(out["pvalue"].to_numpy())
    return out.sort_values(["mp_score", "spearman_r"])

def bh_fdr(pvalues) -> np.ndarray:
    pvalues = np.asarray(pvalues, dtype=float)
    result = np.full(len(pvalues), np.nan)
    ok = np.isfinite(pvalues)
    if ok.sum() == 0:
        return result
    order = np.argsort(pvalues[ok])
    ranked = pvalues[ok][order] * ok.sum() / np.arange(1, ok.sum() + 1)
    ranked = np.minimum.accumulate(ranked[::-1])[::-1]
    values = np.empty(ok.sum())
    values[order] = np.clip(ranked, 0, 1)
    result[ok] = values
    return result


def fdr_label(value: float) -> str:
    if not np.isfinite(value):
        return "FDR=NA"
    if value < 1e-20:
        return "FDR<1e-20"
    if value < 0.001:
        return f"FDR={value:.1e}"
    return f"FDR={value:.3f}".rstrip("0").rstrip(".")

def read_gzip_lines(path: Path) -> list[str]:
    with gzip.open(path, "rt", encoding="utf-8", errors="replace") as handle:
        return [line.rstrip("\n") for line in handle]


def load_spatial_expression():
    raw = RAW_ROOT / "GSE230282_spatial"
    prefix = "GSM7211257_EHU-W3"
    barcodes = read_gzip_lines(raw / f"{prefix}_barcodes.tsv.gz")
    features = pd.read_csv(
        raw / f"{prefix}_features.tsv.gz",
        sep="\t",
        header=None,
        names=["gene_id", "gene_symbol", "feature_type"],
        compression="gzip",
    )
    matrix = mmread(str(raw / f"{prefix}_matrix.mtx.gz")).tocsr()
    positions = pd.read_csv(raw / f"{prefix}_tissue_positions.csv.gz", compression="gzip")
    totals = np.asarray(matrix.sum(axis=0)).ravel()
    totals[totals == 0] = 1
    norm = matrix @ sparse.diags(1e4 / totals)
    norm.data = np.log1p(norm.data)
    spots = positions.merge(pd.DataFrame({"barcode": barcodes}), on="barcode", how="right")
    in_tissue = spots["in_tissue"].eq(1).to_numpy()
    return features, norm[:, in_tissue].tocsr(), spots.loc[in_tissue].reset_index(drop=True)


def score_signature(expr, features: pd.DataFrame, genes: list[str]) -> tuple[np.ndarray, list[str]]:
    present = sorted(set(features.loc[features["gene_symbol"].isin(genes), "gene_symbol"].astype(str)))
    idx = features.index[features["gene_symbol"].isin(present)].to_numpy()
    return np.asarray(expr[idx, :].mean(axis=0)).ravel(), present

def clinical_pairwise() -> tuple[pd.DataFrame, pd.DataFrame]:
    clinical = pd.read_csv(
        RESULT_ROOT / "TCGA_PRAD_clinical_aggressive_state" / "TCGA_PRAD_clinical_MP_scores.tsv",
        sep="\t",
    )
    score = "MP_shortlist_expression_score"
    specifications = {
        "Gleason_group": ["<=6", "7", ">=8"],
        "T_group": ["T2", "T3/T4"],
        "BCR_status": ["No BCR", "BCR"],
    }
    overall_rows = []
    pairwise_rows = []
    for group, order in specifications.items():
        sub = clinical[[score, group]].dropna()
        sub = sub[sub[group].isin(order)]
        samples = [sub.loc[sub[group].eq(level), score] for level in order]
        if len(order) == 3:
            statistic, pvalue = stats.kruskal(*samples)
            test = "Kruskal-Wallis"
        else:
            statistic, pvalue = stats.mannwhitneyu(samples[0], samples[1], alternative="two-sided")
            test = "Mann-Whitney"
        overall_rows.append(
            {
                "clinical_group": group,
                "test": test,
                "statistic": statistic,
                "pvalue": pvalue,
                "n": len(sub),
            }
        )
        for i in range(len(order)):
            for j in range(i + 1, len(order)):
                statistic, pvalue = stats.mannwhitneyu(samples[i], samples[j], alternative="two-sided")
                pairwise_rows.append(
                    {
                        "clinical_group": group,
                        "group_1": order[i],
                        "group_2": order[j],
                        "statistic": statistic,
                        "pvalue": pvalue,
                    }
                )
    overall = pd.DataFrame(overall_rows)
    overall["FDR"] = bh_fdr(overall["pvalue"])
    pairwise = pd.DataFrame(pairwise_rows)
    pairwise["FDR"] = np.nan
    for group, indices in pairwise.groupby("clinical_group").groups.items():
        pairwise.loc[indices, "FDR"] = bh_fdr(pairwise.loc[indices, "pvalue"])
    overall.to_csv(OUT / "TCGA_PRAD_clinical_overall_tests.tsv", sep="\t", index=False)
    pairwise.to_csv(OUT / "TCGA_PRAD_clinical_pairwise_tests.tsv", sep="\t", index=False)
    return clinical, overall, pairwise


def add_bracket(ax, x1, x2, y, height, label):
    ax.plot([x1, x1, x2, x2], [y, y + height, y + height, y], color="#555555", lw=0.75)
    ax.text((x1 + x2) / 2, y + height * 1.15, label, ha="center", va="bottom", fontsize=7.2)

def diverge(value: float) -> str:
    value = max(-0.6, min(0.6, value)) / 0.6
    neg = (176, 201, 222)
    pos = (249, 190, 187)
    white = (255, 255, 255)
    if value >= 0:
        base, t = pos, value
    else:
        base, t = neg, -value
    rgb = tuple(int(white[i] * (1 - t) + base[i] * t) for i in range(3))
    return "#%02x%02x%02x" % rgb

def bh_fdr(pvalues: list[float]) -> np.ndarray:
    pvalues = np.asarray(pvalues, dtype=float)
    result = np.full(len(pvalues), np.nan)
    ok = np.isfinite(pvalues)
    if ok.sum() == 0:
        return result
    order = np.argsort(pvalues[ok])
    ranked = pvalues[ok][order] * ok.sum() / np.arange(1, ok.sum() + 1)
    ranked = np.minimum.accumulate(ranked[::-1])[::-1]
    values = np.empty(ok.sum())
    values[order] = np.clip(ranked, 0, 1)
    result[ok] = values
    return result


def fdr_label(value: float) -> str:
    if not np.isfinite(value):
        return "FDR=NA"
    if value < 1e-20:
        return "FDR<1e-20"
    if value < 0.001:
        return f"FDR={value:.1e}"
    return f"FDR={value:.3f}".rstrip("0").rstrip(".")


def pairwise_tests(clinical: pd.DataFrame) -> pd.DataFrame:
    rows = []
    for group, order, _title in PANELS:
        sub = clinical[[SCORE, group]].dropna()
        sub = sub[sub[group].isin(order)]
        for i in range(len(order)):
            for j in range(i + 1, len(order)):
                a = sub.loc[sub[group].eq(order[i]), SCORE]
                b = sub.loc[sub[group].eq(order[j]), SCORE]
                statistic, pvalue = stats.mannwhitneyu(a, b, alternative="two-sided")
                rows.append(
                    {
                        "clinical_group": group,
                        "group_1": order[i],
                        "group_2": order[j],
                        "statistic": statistic,
                        "pvalue": pvalue,
                        "n_1": len(a),
                        "n_2": len(b),
                    }
                )
    tests = pd.DataFrame(rows)
    tests["FDR"] = np.nan
    for group, indices in tests.groupby("clinical_group").groups.items():
        tests.loc[indices, "FDR"] = bh_fdr(tests.loc[indices, "pvalue"].tolist())
    return tests

            "pdf.fonttype": 42,
            "ps.fonttype": 42,
            "axes.linewidth": 0.75,
            "axes.labelcolor": "#202124",
            "text.color": "#202124",
            "xtick.color": "#202124",
            "ytick.color": "#202124",
        }
    )
    fig, axes = plt.subplots(1, 3, figsize=(9.7, 4.15), sharey=True)
    for panel_idx, (ax, (group, order, title)) in enumerate(zip(axes, PANELS)):
        sub = clinical[[SCORE, group]].dropna()
        sub = sub[sub[group].isin(order)]
        sns.boxplot(
            data=sub,
            x=group,
            y=SCORE,
            order=order,
            ax=ax,
            width=0.55,
            color="#B9CEDD",
            fliersize=0,
            linewidth=0.9,
            boxprops={"edgecolor": "#222222", "linewidth": 0.9},

            whiskerprops={"color": "#222222", "linewidth": 0.9},
            capprops={"color": "#222222", "linewidth": 0.9},
        )
        ax.set_title(title, fontsize=11.5, pad=34)
        ax.set_xlabel("")
        ax.set_ylabel("MP-related expression score" if panel_idx == 0 else "")
        ax.tick_params(axis="x", rotation=18, labelsize=9.5)
        ax.tick_params(axis="y", labelsize=9.5)

        ymin, ymax = sub[SCORE].min(), sub[SCORE].max()
        span = ymax - ymin
        group_tests = tests[tests["clinical_group"].eq(group)]
        if len(order) == 3:
            pairs = [(0, 1), (1, 2), (0, 2)]
            starts = [ymax + span * 0.055, ymax + span * 0.175, ymax + span * 0.315]
            for (x1, x2), y in zip(pairs, starts):
                row = group_tests[group_tests["group_1"].eq(order[x1]) & group_tests["group_2"].eq(order[x2])].iloc[0]
                add_bracket(ax, x1, x2, y, span * 0.025, fdr_label(float(row["FDR"])))
            ax.set_ylim(ymin - span * 0.08, ymax + span * 0.74)
        else:
            row = group_tests.iloc[0]
            add_bracket(ax, 0, 1, ymax + span * 0.085, span * 0.03, fdr_label(float(row["FDR"])))
            ax.set_ylim(ymin - span * 0.08, ymax + span * 0.27)

    sns.despine()
    fig.tight_layout(w_pad=0.75)
    OUT.mkdir(parents=True, exist_ok=True)
    pdf = OUT / "Figure2L_TCGA_PRAD_clinicopathological_pairwise_FDR.pdf"
